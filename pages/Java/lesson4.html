Java Data Structures

Data structures are different ways of organizing data in a computer to use it efficiently. Java provides several built-in data structures through its Collections Framework.

Types of Data Structures in Java

Common Data Structures

Data Structure Interface Implementation Classes Characteristics

Set Set HashSet, TreeSet, LinkedHashSet Unique elements only
Queue Queue PriorityQueue, LinkedList FIFO (First In First Out)
Map Map HashMap, TreeMap, LinkedHashMap Key-value pairs

LinkedList vs ArrayList

Comparison Table

Operation LinkedList ArrayList

HashMap Basics

Creating and Using HashMap

import java.util.HashMap; import java.util.Map; public class HashMapDemo { public static void main(String[] args) { // Creating HashMap HashMap scores = new HashMap<>(); // Adding elements scores.put("John", 95); scores.put("Alice", 88); scores.put("Bob", 92); // Accessing elements System.out.println("John's score: " + scores.get("John")); // Checking if key exists System.out.println("Has Alice? " + scores.containsKey("Alice")); // Checking if value exists System.out.println("Has score 92? import java.util.HashSet; import java.util.Scanner; public class UniqueWordsCounter { private HashSet uniqueWords; public UniqueWordsCounter() { uniqueWords = new HashSet<>(); } public void addWord(String word) { // Convert to lowercase and remove punctuation word = word.toLowerCase().replaceAll("[^a-z]", ""); if (!word.isEmpty()) { uniqueWords.add(word); } } public void processText(String text) { String[] words = text.split("\\s+"); for (String word : words) { addWord(word); } } public int getUniqueWordCount() { return uniqueWords.size(); } public void displayUniqueWords() { System.out.println("Unique words:"); for (String word : uniqueWords) { System.out.println(word); } } public static void main(String[] args) { UniqueWordsCounter counter = new UniqueWordsCounter(); Scanner scanner = new Scanner(System.in); System.out.println("Enter text (empty line to finish):"); String line; while (!(line = scanner.nextLine()).isEmpty()) { counter.processText(line); } System.out.println("\nNumber of unique words: " + counter.getUniqueWordCount()); counter.displayUniqueWords(); scanner.close(); } }

Java Iterator

Iterator is an interface that provides methods to iterate over any Collection. It allows you to traverse through a collection, access or remove elements.

Using Iterator

Basic Iterator Operations

import java.util.*; public class IteratorDemo { public static void main(String[] args) { // Create a collection ArrayList colors = new ArrayList<>(); colors.add("Red"); colors.add("Green"); colors.add("Blue"); colors.add("Yellow"); // Get iterator Iterator it = colors.iterator(); // Traverse forward System.out.println("Traversing forward:"); while (it.hasNext()) { String color = it.next(); System.out.println(color); } // Using ListIterator for bidirectional traversal ListIterator listIt = colors.listIterator(); // Go to end of list while (listIt.hasNext()) { listIt.next(); } // Traverse backward System.out.println("\nTraversing backward:"); while (listIt.hasPrevious()) { String color = listIt.previous(); System.out.println(color); } // Remove elements using iterator it = colors.iterator(); while (it.hasNext()) { String color = it.next(); if (color.equals("Yellow")) { it.remove(); // Safe way to remove while iterating } } System.out.println("\nAfter removal: " + colors); } }

Practice Exercise: Custom Collection with Iterator

import java.util.Iterator; import java.util.NoSuchElementException; class CustomList implements Iterable { private class Node { T data; Node next; Node(T data) { this.data = data; this.next = null; } } private Node head; private int size; public CustomList() { head = null; size = 0; } public void add(T data) { Node newNode = new Node(data); if (head == null) { head = newNode; } else { Node current = head; while (current.next != null) { current = current.next; } current.next = newNode; } size++; } public int size() { return size; } @Override public Iterator iterator() { return new Iterator() { private Node current = head; @Override public boolean hasNext() { return current != null; } @Override public T next() { if (!hasNext()) { throw new NoSuchElementException(); } T data = current.data; current = current.next; return data; } }; } public static void main(String[] args) { CustomList list = new CustomList<>(); list.add("First"); list.add("Second"); list.add("Third"); // Using for-each loop (uses iterator internally) System.out.println("Using for-each loop:"); for (String item : list) { System.out.println(item); } // Using iterator explicitly System.out.println("\nUsing iterator:"); Iterator it = list.iterator(); while (it.hasNext()) { System.out.println(it.next()); } } }

import java.util.ArrayList; import java.util.List; public class NumberProcessor { public static void processNumbers(String... numbers) { List doubles = new ArrayList<>(); // Parse strings to numbers for (String num : numbers) { try { doubles.add(Double.parseDouble(num)); } catch (NumberFormatException e) { System.out.println("Invalid number: " + num); } } if (doubles.isEmpty()) { System.out.println("No valid numbers to process"); return; } // Calculate statistics Double sum = 0.0; Double min = Double.MAX_VALUE; Double max = Double.MIN_VALUE; for (Double num : doubles) { sum += num; min = Math.min(min, num); max = Math.max(max, num); } Double average = sum / doubles.size(); // Display results System.out.println("\nStatistics:"); System.out.printf("Sum: %.2f%n", sum); System.out.printf("Average: %.2f%n", average); System.out.printf("Minimum: %.2f%n", min); System.out.printf("Maximum: %.2f%n", max); // Integer conversions System.out.println("\nInteger conversions:"); for (Double num : doubles) { System.out.printf("%.2f -> %d (rounded)%n", num, Math.round(num)); } } public static void main(String[] args) { System.out.println("Processing numbers:"); processNumbers("10.5", "20.7", "30.2", "invalid", "40.9"); } }
