Python File Handling

File handling is an important part of any programming language. Python has several functions for creating, reading, updating, and deleting files:

  • File operations (open, close, read, write)
  • File modes (read, write, append, etc.)
  • File object methods
  • Context managers (with statement)
# Open a file file = open("example.txt", "r") # "r" for read mode # File modes: # "r" - Read (default) # "w" - Write (overwrites) # "a" - Append # "x" - Create (exclusive) # "b" - Binary mode # "t" - Text mode (default) # "+" - Read and write # Using with statement (recommended) with open("example.txt", "r") as file: content = file.read() # File is automatically closed after block # Check if file exists import os if os.path.exists("example.txt"): print("File exists") # Get file information import os.path print(os.path.getsize("example.txt")) # Size in bytes print(os.path.getmtime("example.txt")) # Last modified time # File object attributes print(file.name) # File name print(file.mode) # File mode print(file.closed) # Check if file is closed # Close file manually (if not using 'with') file.close()

Python Read Files

Python offers several methods to read files:

  • Read entire file
  • Read line by line
  • Read specific number of characters
  • Read into list
# Read entire file with open("example.txt", "r") as file: content = file.read() print(content) # Read line by line with open("example.txt", "r") as file: line = file.readline() while line: print(line.strip()) line = file.readline() # Read all lines into list with open("example.txt", "r") as file: lines = file.readlines() for line in lines: print(line.strip()) # Loop through file directly with open("example.txt", "r") as file: for line in file: print(line.strip()) # Read specific number of characters with open("example.txt", "r") as file: chunk = file.read(5) # Read first 5 characters print(chunk) # Read with encoding with open("example.txt", "r", encoding="utf-8") as file: content = file.read() # Error handling try: with open("nonexistent.txt", "r") as file: content = file.read() except FileNotFoundError: print("File not found") except IOError: print("Error reading file")

Python Write/Create Files

Python can create new files and write content to them:

  • Create new files
  • Write text to files
  • Append to existing files
  • Write multiple lines
# Create and write to file with open("newfile.txt", "w") as file: file.write("Hello, World!") # Write multiple lines with open("newfile.txt", "w") as file: file.write("Line 1\n") file.write("Line 2\n") file.write("Line 3\n") # Write list of lines lines = ["Line 1", "Line 2", "Line 3"] with open("newfile.txt", "w") as file: file.writelines(line + "\n" for line in lines) # Append to file with open("example.txt", "a") as file: file.write("\nNew content appended") # Write with error handling try: with open("output.txt", "w") as file: file.write("Some content") except IOError: print("Error writing to file") # Create file if not exists with open("newfile.txt", "x") as file: file.write("New file created") # Write binary data with open("binary.bin", "wb") as file: file.write(b"Binary data") # Write with specific encoding with open("unicode.txt", "w", encoding="utf-8") as file: file.write("Hello, 世界")

Python Delete Files

Python can delete files and directories using various methods:

  • Delete files
  • Delete directories
  • Check before deleting
  • Error handling
# Delete a file import os if os.path.exists("file.txt"): os.remove("file.txt") print("File deleted") else: print("File does not exist") # Delete directory import shutil # Delete empty directory os.rmdir("empty_folder") # Delete directory and contents shutil.rmtree("folder_with_files") # Safe file deletion try: os.remove("file.txt") except FileNotFoundError: print("File not found") except PermissionError: print("Permission denied") # Move file to recycle bin import send2trash send2trash.send2trash("file.txt") # Delete multiple files import glob for file in glob.glob("*.txt"): try: os.remove(file) print(f"Deleted {file}") except OSError as e: print(f"Error: {file} : {e.strerror}") # Delete files older than X days import time def delete_old_files(path, days): current_time = time.time() for file in os.listdir(path): file_path = os.path.join(path, file) if os.path.isfile(file_path): if os.path.getmtime(file_path) < current_time - days * 86400: os.remove(file_path) print(f"Deleted {file}")
